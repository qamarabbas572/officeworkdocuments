Please use following sudo command if you have ubuntu 16.04

#For Permission 
sudo changeperm 



#For Apache Restart

sudo restartapache
/bin/restartapache

#For MySql restart

sudo restartmysql
/bin/restartmysql

#check MySql status
sudo service mysql status

#To change php versions

sudo set_apache_php

#To Switch Elasticsearch Version7

sudo switch_elasticsearch

sudo stopelasticsearch

sudo startelasticsearch

#To Switch Mysql Server Version

sudo switch_mysql
